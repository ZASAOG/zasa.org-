<script>

/* LOADER */
window.onload = ()=>{
  setTimeout(()=>{
    document.getElementById("loader").style.display="none";
  },2000);
}

/* MENU */
function toggleMenu(){
  let m=document.getElementById("menu");
  m.style.display = m.style.display==="flex"?"none":"flex";
}

function openSection(id){
  document.querySelectorAll(".section").forEach(s=>s.classList.remove("active"));
  document.getElementById(id).classList.add("active");
  document.getElementById("menu").style.display="none";
}

/* LANGUAGE */
let lang="sw";

function toggleLanguage(){
  lang = lang==="sw"?"en":"sw";
  document.querySelectorAll("[data-sw]").forEach(el=>{
    el.innerText = el.getAttribute("data-"+lang);
  });
}

/* ALERT */
function alertLang(type){
  if(type==="maintenance"){
    alert(lang==="sw"?"Huduma ipo kwenye matengenezo":"Service under maintenance");
  }
}

/* JOIN */
function joinInfo(){
  if(lang==="sw"){
    alert("Soma katiba vizuri, pakua fomu, ijaze kisha tuma PDF WhatsApp 0714095979");
  }else{
    alert("Read constitution, download form, fill it and send PDF via WhatsApp 0714095979");
  }
}

/* AI CHATBOT */
function openAI(){
  let q = prompt(lang==="sw" ? "Uliza kuhusu ZASA:" : "Ask about ZASA:");
  if(!q) return;

  let answer = "";

  if(q.toLowerCase().includes("zasa")){
    answer = lang==="sw"
      ? "ZASA ni Zanzibar Aerospace Students Association."
      : "ZASA is Zanzibar Aerospace Students Association.";
  }
  else if(q.toLowerCase().includes("join") || q.toLowerCase().includes("jiunge")){
    answer = lang==="sw"
      ? "Pakua fomu, ijaze kisha tuma WhatsApp 0714095979"
      : "Download the form, fill it and send via WhatsApp 0714095979";
  }
  else if(q.toLowerCase().includes("katiba")){
    answer = lang==="sw"
      ? "Pakua katiba kwenye menu ya Katiba"
      : "Download constitution from Constitution menu";
  }
  else{
    answer = lang==="sw"
      ? "Samahani, sijaelewa swali lako"
      : "Sorry, I didn't understand your question";
  }

  alert(answer);
}
