  let chatbox=document.getElementById("chatbox");
  let msg=input.value.toLowerCase();

  if(!msg) return;

  chatbox.innerHTML += "<div><b>You:</b> "+msg+"</div>";

  let reply="Uliza kuhusu ZASA.";

  if(msg.includes("zasa")){
    reply="ZASA ni taasisi ya wanafunzi wa KIST inayohamasisha elimu ya anga.";

    reply="Pakua fomu, ijaze kisha tuma WhatsApp 0714095979.";
  }

  chatbox.innerHTML += "<div><b>ZASA AI:</b> "+reply+"</div>";
